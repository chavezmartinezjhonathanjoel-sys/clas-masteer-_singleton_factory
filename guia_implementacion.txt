    ## Implementación con método de clase (alternativa)

    Otra forma más simple sin usar `__new__`:

    ```python
    class ConexionBaseDatos:
        _instancia = None
        
        def __init__(self, nombre_bd):
            self.nombre_bd = nombre_bd
            print(f"Inicializando conexión a {nombre_bd}...")
        
        @classmethod
        def obtener_instancia(cls, nombre_bd="base_datos.db"):
            if cls._instancia is None:
                cls._instancia = cls(nombre_bd)
            return cls._instancia
        
        def consultar(self, sql):
            print(f"Ejecutando: {sql}")
            return f"Resultado de {self.nombre_bd}"

Uso:

    conexion1 = ConexionBaseDatos.obtener_instancia("usuarios.db")
    conexion2 = ConexionBaseDatos.obtener_instancia("usuarios.db")

    print(f"¿Son iguales? {conexion1 is conexion2}")  # True
